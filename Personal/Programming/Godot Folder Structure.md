/assets/
    /art/
        /sprites/       # 2D sprites and sprite sheets.
        /textures/      # Textures for 3D models or other uses.
        /models/        # 3D models.
        /tilesets/      # Tilesets for 2D games.
        /fonts/         # Custom fonts used in your game.
    /audio/
        /sfx/           # Sound effects.
        /music/         # Background music.
        /voice/         # Voiceovers.

/scenes/
    /levels/            # Store all level scenes or world areas.
    /ui/                # Store UI scenes like menus, HUDs, and popups.
    /characters/        # Scenes for characters, NPCs, or other game entities.
    /environment/       # Scenes related to the environment, such as background or static elements.

/scripts/
    /characters/        # Scripts related to character behavior, player controls, and NPCs.
    /gameplay/          # Scripts for core gameplay mechanics, game state management, and game logic.
    /ui/                # Scripts for UI behavior, such as menu interactions or HUD updates.
    /utilities/         # General utility scripts, like math functions, helpers, or custom tools.

/resources/
    /materials/         # Materials used in your game, such as shaders or textures applied to objects.
    /prefabs/           # Prefabs or reusable scenes that you might instantiate multiple times.
    /animations/        # Animation files for characters, UI, or other animated elements.
    /theme/             # Theme files for UI, including styleboxes, font settings, and other theme resources.

/addons/                # For any plugins or third-party addons you use in your project.

/doc/                   # Any documentation, notes, or design documents related to your project.

/settings/              # Store configuration files or project settings, such as custom `.cfg` files or `.json` settings.
